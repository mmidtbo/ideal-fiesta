The problem

OpenAPI in Hono is repetitive

Every project ends up with the same fragmented, verbose setup. Fluenth collapses the pipeline.

Without Fluenth

Zod Validation
→
OpenAPI Schema
→
Route Definition
→
Docs Generator
→
Scalar / Swagger
→
Manual sync

With Fluenth

fluenth()
→
Validation + Schema + Routes + Docs
→
Auto docs at /docs
Before
import { Hono } from 'hono'
import { zValidator } from '@hono/zod-validator'
import { OpenAPIRegistry, OpenApiGeneratorV31 } from 'zod-openapi'
import { swaggerUI } from '@hono/swagger-ui'
import { z } from 'zod'

const registry = new OpenAPIRegistry()
const UserSchema = z.object(
  { id: z.string(), name: z.string() }
)
registry.register('User', UserSchema)
registry.registerPath({ ...verbose config... })

const app = new Hono()
app.get('/users/:id',
  zValidator('param', schema),
  (c) => c.json({ id: '1', name: 'Ada' }))

app.get('/openapi.json', (c) => c.json(...))
app.get('/docs', swaggerUI({ url: '/openapi.json' }))
With Fluenth
import { Hono } from 'hono'
import { fluenth, route } from 'fluenth'
import { z } from 'zod'

const app = new Hono()

app.use(fluenth({
  title: 'API',
  version: '1.0.0'
}))

app.get('/users/:id',
  route()
    .params(z.object({ id: z.string() }))
    .response(z.object({
      id: z.string(),
      name: z.string()
    })),
  (c) => c.json({ id: '1', name: 'Ada' })
)
