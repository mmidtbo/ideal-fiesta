Before
import { Hono } from 'hono'
import { zValidator } from '@hono/zod-validator'
import { OpenAPIRegistry, OpenApiGeneratorV31 } from 'zod-openapi'
import { swaggerUI } from '@hono/swagger-ui'
import { z } from 'zod'

const registry = new OpenAPIRegistry()
const UserSchema = z.object(
  { id: z.string(), name: z.string() }
)
registry.register('User', UserSchema)
registry.registerPath({ ...verbose config... })


const app = new Hono()
app.get('/users/:id',
  zValidator('param', schema),
  (c) => c.json({ id: '1', name: 'Ada' }))

app.get('/openapi.json', (c) => c.json(...))
app.get('/docs', swaggerUI({ url: '/openapi.json' }))
With Fluenth
import { Hono } from 'hono'
import { fluenth, route } from 'fluenth'
import { z } from 'zod'

const app = new Hono()

app.use(fluenth({
  title: 'API',
  version: '1.0.0'
}))

app.get('/users/:id',
  route()
    .params(z.object({ id: z.string() }))
    .response(z.object({
      id: z.string(),
      name: z.string()
    })),
  (c) => c.json({ id: '1', name: 'Ada' })
)

// → /docs   → /openapi.json
// auto-generated, type-safe, zero glue ✨

Features

Everything you need. Nothing extra.

⚡
Fluent Integration
Chain documentation alongside your route definitions. No context switching.
🦊
Hono Native
Built specifically for Hono's router and middleware system. Feels native.
📄
Auto OpenAPI
OpenAPI 3.1 spec generated automatically from your route annotations.
✂
Zero Boilerplate
No registry setup, no schema conversion, no separate docs mount.
◈
Scalar Ready
Ships with first-class Scalar UI integration. Drop-in, beautiful docs.
🔷
Type Safe
Full TypeScript support with inferred types from Zod schemas.
🔌
Plug & Play
One middleware call is all you need. Works with existing Hono projects.
☁
Lightweight
Minimal runtime footprint. No heavy dependencies or bundled docs UI bloat.
